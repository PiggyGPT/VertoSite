// ===== MAIN COMPONENT =====

export default function LiquidityFlywheel() {
  const [activeStepId, setActiveStepId] = useState("transactions");
  const [progress, setProgress] = useState(0);

  // Auto-play logic
  useEffect(() => {
    const duration = 5000; // 5 seconds per step
    const interval = 50;
    const steps = duration / interval;
    
    const timer = setInterval(() => {
      setProgress((prev) => {
        if (prev >= 100) {
          const currentIndex = STEPS.findIndex(s => s.id === activeStepId);
          const nextIndex = (currentIndex + 1) % STEPS.length;
          setActiveStepId(STEPS[nextIndex].id);
          return 0;
        }
        return prev + (100 / steps);
      });
    }, interval);

    return () => clearInterval(timer);
  }, [activeStepId]);

  // Handler for manual click
  const handleStepClick = (id: string) => {
    setActiveStepId(id);
    setProgress(0); // Reset progress on manual interaction
  };

  const renderActiveDashboard = () => {
    switch (activeStepId) {
      case "transactions": return <TransactionView />;
      case "fees": return <YieldView />;
      case "liquidity": return <LiquidityView />;
      case "compliance": return <ComplianceView />;
      default: return <TransactionView />;
    }
  };

  return (
    <section className="relative w-full py-24 bg-slate-950 text-slate-50 overflow-hidden">
      {/* Background Ambience */}
      <div className="absolute inset-0 pointer-events-none">
         <div className="absolute top-0 right-0 w-[500px] h-[500px] bg-blue-900/10 rounded-full blur-[100px]" />
         <div className="absolute bottom-0 left-0 w-[500px] h-[500px] bg-emerald-900/10 rounded-full blur-[100px]" />
      </div>

      <div className="container mx-auto px-4 relative z-10">
        
        {/* Header */}
        <div className="mb-16 md:text-center max-w-2xl mx-auto">
          <motion.div
            initial={{ opacity: 0, y: 10 }}
            whileInView={{ opacity: 1, y: 0 }}
            viewport={{ once: true }}
            className="flex items-center gap-2 text-blue-400 font-mono text-xs uppercase tracking-widest mb-4 justify-center"
          >
             <RefreshCw className="w-3 h-3" />
             <span>The Growth Engine</span>
          </motion.div>
          
          <motion.h2 
            initial={{ opacity: 0, y: 10 }}
            whileInView={{ opacity: 1, y: 0 }}
            viewport={{ once: true }}
            className="text-3xl md:text-4xl lg:text-5xl font-bold text-white mb-6"
          >
            A Self-Reinforcing <span className="text-transparent bg-clip-text bg-gradient-to-r from-blue-400 to-emerald-400">Liquidity Loop</span>
          </motion.h2>
          
          <motion.p
             initial={{ opacity: 0, y: 10 }}
             whileInView={{ opacity: 1, y: 0 }}
             viewport={{ once: true }}
             transition={{ delay: 0.1 }}
             className="text-slate-400 text-lg"
          >
            Real-time visualization of how Albor transforms transaction volume into deep, compliant bank liquidity.
          </motion.p>
        </div>

        {/* Layout: Split View on Desktop, Stacked on Mobile */}
        <div className="grid grid-cols-1 lg:grid-cols-12 gap-8 lg:gap-12 items-start">
          
          {/* Left Column: Interactive Steps */}
          <div className="lg:col-span-5 flex flex-col gap-4">
            <LayoutGroup>
              {STEPS.map((step) => {
                const isActive = activeStepId === step.id;
                return (
                  <motion.div
                    key={step.id}
                    onClick={() => handleStepClick(step.id)}
                    className={`
                      relative p-6 rounded-xl cursor-pointer border transition-all duration-500 overflow-hidden
                      ${isActive 
                        ? "bg-slate-900 border-slate-700 shadow-2xl" 
                        : "bg-slate-950/50 border-transparent hover:bg-slate-900/50 hover:border-slate-800"
                      }
                    `}
                  >
                    {/* Progress Bar Background */}
                    {isActive && (
                      <motion.div 
                        className="absolute bottom-0 left-0 h-[2px] bg-gradient-to-r from-blue-500 to-emerald-500 z-20"
                        style={{ width: `${progress}%` }}
                      />
                    )}

                    <div className="flex items-start justify-between relative z-10">
                      <div>
                        <div className={`text-xs font-mono mb-2 uppercase tracking-wider ${isActive ? "text-blue-400" : "text-slate-500"}`}>
                           0{STEPS.indexOf(step) + 1} / {step.subtitle}
                        </div>
                        <h3 className={`text-xl font-semibold mb-2 ${isActive ? "text-white" : "text-slate-400"}`}>
                          {step.title}
                        </h3>
                        <AnimatePresence>
                          {isActive && (
                            <motion.p 
                              initial={{ opacity: 0, height: 0 }}
                              animate={{ opacity: 1, height: "auto" }}
                              exit={{ opacity: 0, height: 0 }}
                              className="text-slate-400 text-sm leading-relaxed"
                            >
                              {step.description}
                            </motion.p>
                          )}
                        </AnimatePresence>
                      </div>
                      
                      {/* Active Indicator Arrow */}
                      {isActive && (
                        <motion.div
                           layoutId="active-arrow"
                           className="text-emerald-400"
                        >
                           <ArrowUpRight className="w-5 h-5" />
                        </motion.div>
                      )}
                    </div>
                  </motion.div>
                );
              })}
            </LayoutGroup>
          </div>

          {/* Right Column: The "Terminal" Dashboard */}
          <div className="lg:col-span-7 sticky top-24">
             <div className="relative">
                {/* Device Frame / Glow */}
                <div className="absolute -inset-1 bg-gradient-to-b from-blue-500/20 to-purple-500/20 rounded-2xl blur-lg opacity-50" />
                
                {/* Main Dashboard Container */}
                <div className="relative bg-slate-900 rounded-xl border border-slate-700 shadow-2xl overflow-hidden h-[400px] flex flex-col">
                   
                   {/* Terminal Header */}
                   <div className="h-12 border-b border-slate-800 bg-slate-900 flex items-center justify-between px-4">
                      <div className="flex items-center gap-2">
                         <div className="w-3 h-3 rounded-full bg-red-500/20 border border-red-500/50" />
                         <div className="w-3 h-3 rounded-full bg-amber-500/20 border border-amber-500/50" />
                         <div className="w-3 h-3 rounded-full bg-emerald-500/20 border border-emerald-500/50" />
                      </div>
                      <div className="flex items-center gap-2 px-3 py-1 rounded-full bg-slate-800 border border-slate-700">
                         <Lock className="w-3 h-3 text-slate-400" />
                         <span className="text-[10px] text-slate-400 font-mono uppercase">Albor_Mainnet_Live</span>
                      </div>
                      <MoreHorizontal className="w-4 h-4 text-slate-600" />
                   </div>

                   {/* Dashboard Content Area */}
                   <div className="flex-1 p-6 relative">
                      <AnimatePresence mode="wait">
                         <motion.div
                           key={activeStepId}
                           initial={{ opacity: 0, y: 10, scale: 0.98 }}
                           animate={{ opacity: 1, y: 0, scale: 1 }}
                           exit={{ opacity: 0, y: -10, scale: 0.98 }}
                           transition={{ duration: 0.3 }}
                           className="h-full w-full"
                         >
                            {renderActiveDashboard()}
                         </motion.div>
                      </AnimatePresence>
                   </div>

                   {/* Terminal Footer */}
                   <div className="h-8 border-t border-slate-800 bg-slate-950 flex items-center justify-between px-4 text-[10px] text-slate-500 font-mono">
                      <div className="flex items-center gap-3">
                         <span>SYNC: <span className="text-emerald-500">OK</span></span>
                         <span>LATENCY: 12ms</span>
                      </div>
                      <div className="flex items-center gap-2">
                         <div className="w-1.5 h-1.5 rounded-full bg-emerald-500 animate-pulse" />
                         <span>LIVE DATA FEED</span>
                      </div>
                   </div>

                </div>
             </div>
          </div>

        </div>
      </div>
    </section>
  );
}
